    AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_PIOA;

    AT91C_BASE_PIOA->PIO_IDR = 0xFFFFFFFF;

    AIC_EnableIT(AT91C_ID_PIOA);
