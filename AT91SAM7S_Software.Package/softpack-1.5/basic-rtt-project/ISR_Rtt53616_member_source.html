void ISR_Rtt(void)
{
    unsigned int status;

    // Get RTT status
    status = RTT_GetStatus(AT91C_BASE_RTTC);

    // Time has changed, refresh display
    if ((status & AT91C_RTTC_RTTINC) == AT91C_RTTC_RTTINC) {

        RefreshDisplay();
    }

    // Alarm
    if ((status & AT91C_RTTC_ALMS) == AT91C_RTTC_ALMS) {

        alarmed = 1;
        RefreshDisplay();
    }
}
