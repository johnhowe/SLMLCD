void DumpMemorySDSPI(const unsigned char *pBuffer, unsigned int size)
{
    unsigned int i, j;

    for (i=0; i < (size / 16); i++) {

        printf("0x%08X: ", i*16);
        for (j=0; j < 4; j++) {

            printf("%08X ", ((unsigned int *) pBuffer)[i*4+j]);
        }
        printf("\n\r");
    }
}
